```php
<?php
	
$a = 4;
$b = 2;

// 1. Adição
echo $a + $b;  # Retorna: 6


// 2. Subtração
echo $a - $b;  # 2


// 3. Multiplicação
echo $a * $b;  # 8


// 4. Divisão (retorna valor decimal real)
echo $a / $b;  # 2


// 6. Módulo (retorna o resto da divisão)
echo $a % $b;  # 0 


// 7. Exponencial
echo $a ** $b; # 16


// 8. Negação (retorna o oposto)
echo -$a;  # -4
```
